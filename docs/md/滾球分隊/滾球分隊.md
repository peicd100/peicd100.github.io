        function calculateGroups(n) {
            let resultText = "";
            if(n>1000){
                return "請不要輸入大於1000的數字";
            }

            let w = n;

            while (w--) {
                let p = 0;
                resultText += `共 ${n} 人時：\n`;
                // 計算分組
                for (let a = Math.floor(n / 2); a >= 0; a--) {
                    let b = n - a * 2;
                    if (b % 3 === 0) {
                        b /= 3;
                        if ((a + b) % 2 === 0) {
                            p = 1;
                            resultText += `${a} 組二人，${b} 組三人，共 ${(a + b) / 2} 個場地\n`;
                        }
                    }
                    
                }
                if(p==0){
                    resultText += `無法分隊\n`;
                }

                n--;
                resultText += `--------------------------------------------------------\n`;
            }

            return resultText;
        }

        function updateResult() {
            const n = parseInt(document.getElementById('numberInput').value, 10);

            if (isNaN(n) || n <= 0) {
                document.getElementById('result').textContent = '請輸入有效的正整數';
                return;
            }

            const result = calculateGroups(n);
            document.getElementById('result').textContent = result;
        }

        document.getElementById('numberInput').addEventListener('input', updateResult);
    </script>
